CREATE TABLE `permit_licence` (
  `PERMIT_NO` int(11) NOT NULL AUTO_INCREMENT,
  `PERMIT_TYPE` varchar(20) NOT NULL,
  `CLUB_NAME` varchar(45) NOT NULL,
  `APPLICANT_NAME` varchar(45) NOT NULL,
  `TRACK_NAME` varchar(45) NOT NULL,
  `PERMIT_START_DATE` datetime NOT NULL,
  `PERMIT_END_DATE` datetime NOT NULL,
  PRIMARY KEY (`PERMIT_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
